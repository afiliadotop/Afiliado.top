package-lock.json

.DS_Store
Thumbs.db

# Ignorar configurações locais
.env

# Ignorar pastas de build ou cache
dist/
build/
.cache/

# Ignorar uploads temporários
tmp/
temp/

# Ignorar logs
*.log

# Ignorar pastas do editor
